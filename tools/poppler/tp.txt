a. dans le fichier backend/route/auth.js :

Inscription utilisateur : POST /api/auth/signup

Cette route permet de créer un nouvel utilisateur dans la base de données.

Quand l'utilisateur remplit le formulaire Signup dans React, le frontend envoie une requête :




Fonctions utilisées :

    req.body
    bcrypt.hash()
    db.query()

Étapes :

 1. récupérer name, email, password
 2. vérifier si email existe
 3. crypter mot de pass
 4. enregistrer utilisateur
POST /api/auth/login sert à connecter un utilisateur.

Étapes :

 1. Chercher utilisateur dans MySQL avec son email (db.query() / pool.query() + SELECT ...
    WHERE email=?).
 2. Vérifier le mot de passe : comparer le password saisi avec password_hash stocké
    (bcrypt.compare()).
 3. Créer un token JWT si tout est correct (jwt.sign()), puis renvoyer le token au frontend

const router = require("express").Router();

const bcrypt = require("bcrypt");

const jwt = require("jsonwebtoken");

const pool = require("../db");

require("dotenv").config();




/**

* POST /api/auth/signup

* Body: { name, email, password, age?, cin? }

*/

router.post("/signup", async (req, res) => {

 try {

  // 1) récupérer les données

  const { name, email, password, age = null, cin = null } = req.body;




  // 2) vérifier champs obligatoires

  if (!name || !email || !password) {

      return res.status(400).json({ message: "Missing required fields" });

  }




  // 3) vérifier si email existe déjà
      const [existing] = await pool.query("SELECT id FROM users WHERE email = ?", [email]);

      if (existing.length > 0) {

          return res.status(409).json({ message: "Email already exists" });

      }




      // 4) crypter (hasher) le mot de passe

      const password_hash = await bcrypt.hash(password, 10);




      // 5) enregistrer l'utilisateur dans MySQL

      const [result] = await pool.query(

          "INSERT INTO users (name, email, password_hash, age, cin) VALUES (?, ?, ?, ?, ?)",

          [name, email, password_hash, age, cin]

      );




      // 6) renvoyer réponse (sans mot de passe)

      return res.status(201).json({

          message: "User created successfully",

          user: { id: result.insertId, name, email, age, cin },

      });

 } catch (err) {

      return res.status(500).json({ message: "Server error" });

 }

});




/**

* POST /api/auth/login
* Body: { email, password }

*/

router.post("/login", async (req, res) => {

 try {

  // 1) récupérer email + password

  const { email, password } = req.body;




  // vérifier champs

  if (!email || !password) {

      return res.status(400).json({ message: "Missing fields" });

  }




  // 2) chercher utilisateur dans MySQL

  const [rows] = await pool.query("SELECT * FROM users WHERE email = ?", [email]);

  if (rows.length === 0) {

      return res.status(401).json({ message: "Invalid credentials" });

  }




  const user = rows[0];




  // 3) vérifier le mot de passe (compare password avec password_hash)

  const ok = await bcrypt.compare(password, user.password_hash);

  if (!ok) {

      return res.status(401).json({ message: "Invalid credentials" });

  }
      // 4) créer token JWT

      const token = jwt.sign(

       { id: user.id, email: user.email },   // payload

       process.env.JWT_SECRET,                // secret

       { expiresIn: "2h" }             // durée

      );




      // 5) renvoyer token + infos user (sans password_hash)

      return res.json({

       token,

       user: {

            id: user.id,

            name: user.name,

            email: user.email,

            age: user.age,

            cin: user.cin,

       },

      });

 } catch (err) {

      return res.status(500).json({ message: "Server error" });

 }

});

c.backend/middleware/auth.js:

Ce middleware sert à protéger les routes.Il vérifie que l'utilisateur possède un token JWT
valide.Quand un utilisateur se connecte :POST /api/auth/login le serveur crée un token JWT.
const jwt = require("jsonwebtoken");

require("dotenv").config();




module.exports = function auth(req, res, next) {




 // récupérer le header Authorization

 const authHeader = req.headers.authorization;




 // vérifier si le header existe

 if (!authHeader) {

     return res.status(401).json({

      message: "No token provided"

     });

 }




 // le header doit être : Bearer TOKEN

 const parts = authHeader.split(" ");




 if (parts.length !== 2 || parts[0] !== "Bearer") {

     return res.status(401).json({

      message: "Invalid token format"

     });

 }




 const token = parts[1];
 try {




     // vérifier le token

     const decoded = jwt.verify(

      token,

      process.env.JWT_SECRET

     );




     // ajouter les infos utilisateur dans req.user

     req.user = decoded;



     // continuer vers la route

     next();



 } catch (error) {




     return res.status(401).json({

      message: "Invalid token"

     });




 }




};
